```cpp
#include<cstdio> 
#include <cmath>  
using namespace std;  
double C(int n,int m)  
{  
	double sum1=1,sum2=1;  
    for(int i=0;i<m;i++)  
        sum1*=(n-i);   
    int k=m;  
    for(int i=1;i<=m;i++)      
        sum2*=k,k--;  
    return sum1/sum2;  
}  
int main()  
{  
    int T,n,m; 
    double x,sum;    
    scanf("%d",&T);  
      
    while(T--)  
    {  
		sum=0;
  		scanf("%d%d%lf",&n,&m,&x);  
        for(int k=m;k<=n;k++)  
            sum+=C(n,k)*pow(x,k)*pow(1-x,n-k);   
        printf("%.3lf\n",sum); 
    }  
    return 0;  
}  
```
