```cpp

typedef long long ll;
int ex_gcd(int a,int b,int &x,int &y)
{
    int gcd;
    if(!b)
    {
        x=1,y=0;
        return a;
    }
    else
    {
        gcd = ex_gcd(b,a%b,y,x);
        y-=x*(a/b);
    }
    return gcd;
}

```
